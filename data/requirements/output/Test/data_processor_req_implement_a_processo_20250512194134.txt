/**
 * Processor for string data, providing functionality to process, calculate total length, and filter by length.
 */
public class ProcessorData {

    /**
     * Processes a list of strings and returns processed data objects with the string value and its length.
     * 
     * @param data The list of strings to process
     * @return A list of processed data objects
     */
    public List<ProcessedStringData> processStrings(List<String> data) {
        List<ProcessedStringData> results = new ArrayList<>();
        for (String item : data) {
            ProcessedStringData processedItem = new ProcessedStringData(item, item.length());
            results.add(processedItem);
        }
        return results;
    }

    /**
     * Calculates the total length of all strings in a list.
     * 
     * @param data The list of strings to calculate length for
     * @return The total length of all strings
     */
    public int calculateTotalStringLength(List<String> data) {
        return data.stream().mapToInt(String::length).sum();
    }

    /**
     * Filters strings by a minimum length threshold.
     * 
     * @param data       The list of strings to filter
     * @param minLength The minimum length threshold
     * @return A filtered list of strings
     */
    public List<String> filterByLength(List<String> data, int minLength) {
        return data.stream().filter(item -> item.length() >= minLength).collect(Collectors.toList());
    }

    /**
     * Data class representing processed string data with the string value and its length.
     */
    public static class ProcessedStringData {
        private final String value;
        private final int length;

        public ProcessedStringData(String value, int length) {
            this.value = value;
            this.length = length;
        }

        public String getValue() {
            return value;
        }

        public int getLength() {
            return length;
        }

        @Override
        public String toString() {
            return "ProcessedStringData{" +
                    "value='" + value + '\'' +
                    ", length=" + length +
                    '}';
        }
    }
}

public static void main(String[] args) {
    ProcessorData processor = new ProcessorData();
    List<String> sampleData = List.of("Test", "Sample", "Java", "Programming");

    // Process Strings
    List<ProcessorData.ProcessedStringData> processed = processor.processStrings(sampleData);
    System.out.println("Processed Data:");
    processed.forEach(System.out::println);

    // Calculate Total Length
    int totalLength = processor.calculateTotalStringLength(sampleData);
    System.out.println("\nTotal Length of All Strings: " + totalLength);

    // Filter by Length
    List<String> filtered = processor.filterByLength(sampleData, 4);
    System.out.println("\nFiltered Strings (Length >= 4):");
    filtered.forEach(System.out::println);
}
